GDB - Gnu Debug

TCP Port 1234 (default)

Packet (text)
	Header			1		"$"
	Data			n
	Footer			1		"#"
	Checksum		2		"FF"  // ASCII 2 character of Data bytes summed mod 256 (hexadecimal)

Acknowledge			1		"+"
Retransmit			1		"-"

Commands
	!			Enable Extended Mode
	?			Get Last Error
	D			Disconnect
	c			Continue
	g			Read General Registers
	G			Write General Registers
	H			Set Thread
	i			Step One Cycle
	k			Kill Thread
	m			Read Memory					m[Address],[Length]
	M			Write Memory				M[Address],[Length]:[Data]
	p			Read Register				p[Register]
	P			Write Register				P[Register]=[Value]
	s			Step One Instruction		S[Address]
	t			Search						t[Address]:[Pattern],[Mask]
	X			Write Memory Binary			X[Address],[Length]:[Data]
	z			Remove Breakpoint			z[Type],[Address],[Length]
	Z			Add Breakpoint				Z[Type],[Address],[Length]
