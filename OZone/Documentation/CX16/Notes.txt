Memory Map
	0x9f00-0x9f1f		Audio Controller
	0x9f20-0x9f3f		Video Controller
		0x9f20		VRAM Address Low
		0x9f21		VRAM Address Mid
		0x9f22		VRAM Address High
			Address Increment		4 bits
			Decrement Mode			1 bit
			Reserved				2 bits
			VRAM Address High		1 bit
		0x9f23		VRAM Data 0
		0x9f24		VRAM Data 1
		0x9f25		Control
			Reset					1 bit
			Reserved				5 bits
			DCSEL					1 bit
			Address Select			1 bit
		0x9f26		Interrupt Enable
		0x9f27		Interrupt Status
		...
		0x9f30		Horizontal Scroll 0 Low
		0x9f31		Horizontal Scroll 0 High
		0x9f32		Vertical Scroll 0 Low
		0x9f33		Vertical Scroll 0 High
		...
		0x9f37		Horizontal Scroll 1 Low
		0x9f38		Horizontal Scroll 1 High
		0x9f39		Vertical Scroll 1 Low
		0x9f3a		Vertical Scroll 1 High
	0x9f40-0x9f41		Yamaha YM2151 OPL Chip
		0x9f40		Address
		0x9f41		Data
	0x9f60-0x9f6f		Interface Controller 0
		0x9f60	Data B
			Reserved		5 bits
			ROM Bank		3 bits			// 16KiB ROM Bank (0xC000-0xFFFF)
		0x9f61	Data A
			RAM Bank		8 bits			// 8KiB RAM Bank (0xA000-0xBFFF)  1 = Kernel
		0x9f62	Mode B
			Reserved		5 bits
			ROM Bank		3 bits
		0x9f63	Mode A
			RAM Bank		8 bits
	0x9f70-0x9f7f		Interface Controller 1
		0x9f70	Data B
		0x9f71	Data A
			Reserved				2 bits
			Joypad 2				1 bit
			Joypad Clock			1 bit
			Joypad 1				1 bit
			Joypad Latch			1 bit
			Reserved				1 bit
			Keyboard Clock		1 bit
			Keyboard Data		1 bit
		0x9f72	Mode B						// 0 = Input, 1 = Output
		0x9f73	Mode A						// 0 = Input, 1 = Output
			Reserved				2 bits
			Joypad 2				1 bit
			Joypad Clock			1 bit
			Joypad 1				1 bit
			Joypad Latch			1 bit
			Reserved				1 bit
			Keyboard Clock		1 bit
			Keyboard Data		1 bit
		0x9f7c	Control
		0x9f7d	Interrupt Flags
			Interrupt				1 bit
			Timer 1					1 bit
			Timer 2					1 bit
			CB1						1 bit
			CB2						1 bit
			Shift						1 bit
			CA1						1 bit
			CA2						1 bit
		0x9f7e	Interrupt Enable
			Set						1 bit
			Timer 1					1 bit
			Timer 2					1 bit
			CB1						1 bit
			CB2						1 bit
			Shift						1 bit
			CA1						1 bit
			CA2						1 bit
	0x9f80-0x9f9f		Real Time Clock

VERA Memory Map (KERNAL Default)
	0x000000	Bitmap
	0x013000	Sprite Bitmaps
	0x01B000	Character Map
	0x01F000	Character Set
	0x01F9C0	Registers

Old VERA Memory Map (KERNEL Default)
	0x000000	Character Map
